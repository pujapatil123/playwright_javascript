cd path/to/my-local-project # Navigate to your project folder
git init                     # Initialize a local Git repository
git add .                    # Stage all files for the first commit
git commit -m "Initial commit" # Create the first commit
git remote add origin [URL]  # Connect to the remote GitHub repository
git branch -M main           # Rename local branch to 'main'
git push -u origin main      # Push the code to GitHub and set the upstream
